# Substitution Cipher
# scratch work by John Reyner

# I looked up a basic code for encryption/decryption online to replicate as practice.
# To add in some personal work I added in a means to accept a non-case sensitive input of 'Encrypt' or 'Decrypt' instead of just the numeric inputs.
# There also is a means to set any integer as an encryption code.

result = ''
choice = ''
message = ''

while choice != '0':
    choice = input("\n Choose to encrypt or decrypt.\n 1 to encrypt, 2 to decrypt, 0 to cancel.")

    if choice.lower() == 'encrypt':
        choice = '1'

    if choice.lower() == 'decrypt':
        choice = '2'

    if choice == '1':
        message = input('\nEnter message to encrypt: ')
        iseed = input('\nInput an integer to use as a seed.')
        seed = int(iseed)
        for i in range(0, len(message)):
            result = result + chr(ord(message[i]) - seed)

        print(result + '\n\n')
        result = ''

    elif choice == '2':
        message = input('\nEnter message to decrypt: ')
        iseed = input('\nInput the integer that was used as the seed.')
        seed = int(iseed)
        for i in range(0, len(message)):
            result = result + chr(ord(message[i]) + seed)

        print(result + '\n\n')
        result = ''

    elif choice != '0':
        print('Invalid input, please try again. \n\n')
